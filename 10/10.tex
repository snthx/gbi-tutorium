\begin{frame}{Regeln im Hoare-Kalkül}

	\begin{block}{HT-E: Verstärkung der Vorbedingung und Abschwächung der Nachbedingung}
		Es gelte $P' \vdash P$ und $Q\vdash Q'$.\\[18pt]
		Dann gilt: \begin{tabular}{c}
						$\set{P} \, S \, \set{Q}$\\
						\midrule
						$\set{P'} \, S \, \set{Q'}$
					\end{tabular}
	\end{block}
